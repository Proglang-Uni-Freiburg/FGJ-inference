class Pair <X extends Object <>, Y extends Object<>> extends Object<> {
    X fst;
    Y snd;

    setfst(fst) {
        return new Pair(fst, this.snd);
    }

    swap() {
        return new Pair(this.snd, this.fst);
    }

    setboth(fst, snd) {
        return new Pair(new Pair(fst, this.fst).fst, new Pair(snd, this.snd).fst);
    }
}

new Object()
